local __namecall
__namecall = function(...)
    local Arguments = {...}
    local self = Arguments[1]
    local Method = getnamecallmethod()

    if not checkcaller()
        and (Method == "FireServer" or Method == "fireServer")
        and self.Name == "CharacterSoundEvent"
    then
         local arg3 = Arguments[3]

         if type(arg3) == "string" then
            local lower = string.lower(arg3)

           if lower:find("catch", 1, true) then
                return oldNamecall(...)
            elseif #arg3 > 30 then
                return
            elseif lower:find("error", 1, true) then
                return
            end
        end

        if type(arg3) == "table" and foundCount < 3 then
            if not savedTables[1] then
                savedTables[1] = arg3
                foundCount += 1
            elseif not savedTables[2] and not rawequal(arg3, savedTables[1]) then
                savedTables[2] = arg3
                foundCount += 1
            elseif not savedTables[3] and not rawequal(arg3, savedTables[1]) and not rawequal(arg3, savedTables[2]) then
                savedTables[3] = arg3
                foundCount += 1
            end
        end
    end

    return oldNamecall(...)
end

local function bump19(slotTable, label)
    local inner = slotTable[1]
    local current = inner[19]
    if typeof(current) ~= "number" then
        return slotTable
    end

    local stored = callCounters[slotTable] or current

    if stored < 1000 then
        stored = stored + 1
        callCounters[slotTable] = stored
        inner[19] = stored
    end

    return slotTable
end
